    Retorna um array dizendo se esta lotado e quantas vagas livres ainda restam

    # File servidor.rb, line 15
def estado
        cheio = @atual >= @vagas
        mensagem = if cheio 
                "Lotado." 
        else
                "Ha #{@vagas - @atual} vagas livres."
        end
        [cheio, mensagem]
end

def sai_carro
        if @atual > 0
                @atual -= 1  
                retorna = "Um carro saiu."
        else
                retorna = "Nao ha carros para sair."
        end
        print retorna + "\n"
        print "Status: #{estado[1]}\n"
        retorna
end
